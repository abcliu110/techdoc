# 消费赠券企业级架构实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 POS 结账和线上结账场景下，实现可靠、可幂等、可补偿的消费赠券闭环，满足企业级可靠性要求。

**Architecture:** 三层架构：
- **POS 侧**（本地）：ActiveMQ 本地缓冲 + 本地消费者转发 CRM + Session/Round/Event 三表本地状态机 + 定时补偿兜底
- **CRM 侧**（云端）：幂等任务表（复用 CrmConsumeBenefitTask）+ 规则快照 + 已使用券策略
- **降级/容灾**：ActiveMQ 不可用时降级 Feign 直调；消费失败进 DLQ；补偿任务同时扫 PENDING 和 FAILED

**Tech Stack:** Java 17, MyBatis-Plus, ActiveMQ, Feign, Spring Schedule, MySQL

```sql
CREATE TABLE crm_consume_coupon_session (
    lid         BIGINT PRIMARY KEY,
    mid         BIGINT NOT NULL,
    sid         BIGINT NOT NULL,
    card_lid    BIGINT NOT NULL,
    card_no     VARCHAR(32),
    origin_lifecycle_id  VARCHAR(64),  -- POS 清台/round ID
    source      VARCHAR(16) NOT NULL DEFAULT 'POS',  -- POS / ORDER
    status      VARCHAR(16) NOT NULL DEFAULT 'ACTIVE', -- ACTIVE / COMPLETED / CLOSED
    created_time DATETIME,
    updated_time DATETIME,
    deleted     TINYINT DEFAULT 0,
    INDEX idx_mid_sid (mid, sid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 轮次：每个赠券操作（含 grant / revoke）
CREATE TABLE crm_consume_coupon_round (
    lid             BIGINT PRIMARY KEY,
    session_lid     BIGINT NOT NULL,
    mid             BIGINT NOT NULL,
    sid             BIGINT NOT NULL,
    card_lid        BIGINT NOT NULL,
    card_no         VARCHAR(32),
    bill_lid        BIGINT,                  -- 账单 LID（幂等键字段）
    out_order_id    VARCHAR(64),             -- 外部订单号（幂等键字段）
    event_type      VARCHAR(16) NOT NULL,    -- GRANT / REVOKE
    status          VARCHAR(32) NOT NULL DEFAULT 'PENDING',
    -- 幂等键: mid + sid + source + bill_lid + event_type + out_order_id + card_lid
    business_key    VARCHAR(128) NOT NULL,
    crm_task_lid    VARCHAR(64),             -- CRM 任务 LID（回调后更新）
    rule_lid        BIGINT,                  -- 命中的规则 LID
    rule_snapshot   JSON,                    -- 规则快照（补偿重试用）
    grant_amount    DECIMAL(10,2),           -- 赠券金额（快照）
    error_message   VARCHAR(256),            -- 最近失败原因
    retry_count     INT DEFAULT 0,
    next_retry_time DATETIME,
    created_time    DATETIME,
    updated_time    DATETIME,
    deleted         TINYINT DEFAULT 0,
    UNIQUE KEY uk_business_key (business_key, deleted),
    INDEX idx_status_retry (status, next_retry_time),
    INDEX idx_session (session_lid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 事件：操作流水（可排查性）
CREATE TABLE crm_consume_coupon_event (
    lid             BIGINT PRIMARY KEY,
    round_lid       BIGINT NOT NULL,
    event_type      VARCHAR(16) NOT NULL,  -- GRANT / REVOKE / RETRY / COMPENSATION / CALLBACK
    event_status    VARCHAR(16) NOT NULL,  -- SUCCESS / FAILED / PENDING
    request_payload TEXT,                  -- 发出时的请求体
    response_payload TEXT,                 -- CRM 返回体
    error_message   VARCHAR(256),
    created_time    DATETIME,
    INDEX idx_round (round_lid),
    INDEX idx_created (created_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

```java
// CrmConsumeCouponActiveMQConsumer.java
@JmsListener(destination = "CRM_CONSUME_COUPON_GRANT_QUEUE")
public void onMessage(Message msg, Session session) {
    ConsumeCouponGrantMsg grantMsg = parse(msg);
    String businessKey = grantMsg.getBusinessKey();

    try {
        // 1. 更新 round 为 GRANT_PENDING（幂等，已有则跳过）
        CrmConsumeCouponRound round = updateRoundPending(grantMsg);

        // 2. 调 CRM（带超时）
        NmsResult<String> result;
        try {
            result = crmConsumeCouponFeign.consumerGiveCoupon(toDTO(grantMsg));
        } catch (Exception e) {
            // 3. ActiveMQ 不可用时降级 Feign（不依赖本地 MQ）
            log.warn("ActiveMQ 调用失败，降级 Feign 直调 CRM", e);
            result = crmCardOpServiceFeign.consumerGiveCoupon(toDTO(grantMsg));
        }

        if (result.isSuccess()) {
            // 4. 成功：更新 round 为 GRANTED，记录 CRM 返回
            roundService.markGranted(round.getLid(), extractTaskLid(result));
            eventService.logSuccess(round.getLid(), "CALLBACK", result);
            session.commit();
        } else {
            // 5. 业务失败：round 进 FAILED，等待补偿
            roundService.markFailed(round.getLid(), result.getMsg());
            eventService.logFailed(round.getLid(), "CALLBACK", result.getMsg());
            session.rollback(); // 让 MQ 重试
        }
    } catch (Exception e) {
        log.error("消费赠券失败，businessKey={}", businessKey, e);
        eventService.logFailed(round.getLid(), "CALLBACK", e.getMessage());
        session.rollback(); // MQ 重试
    }
}
```

```java
// CrmConsumeCouponCompensationJob.java
@Scheduled(fixedDelayString = "${coupon.compensation.interval:60000}")
public void compensate() {
    log.info("消费赠券补偿任务开始...");
    int limit = 50;
    List<CrmConsumeCouponRound> candidates = roundService.scanPendingOrFailed(limit);

    for (CrmConsumeCouponRound round : candidates) {
        try {
            // 指数退避检查
            if (round.getNextRetryTime() != null && LocalDateTime.now().isBefore(round.getNextRetryTime())) {
                continue; // 未到重试时间，跳过
            }
            // 最大重试次数（如 10 次），超过则标记为 RETRYABLE（需人工处理）
            if (round.getRetryCount() >= 10) {
                roundService.update().lambdaSet()
                    .eq(CrmConsumeCouponRound::getStatus, "RETRYABLE")
                    .eq(CrmConsumeCouponRound::getLid, round.getLid()).update();
                log.warn("消费赠券 round {} 超过最大重试次数，进入人工处理", round.getLid());
                alertService.sendMaxRetryAlert("consume-coupon", round.getBusinessKey());
                continue;
            }
            // 幂等重试：business_key 不变，重试只补发失败部分
            CrmConsumeCouponLocalService retryResult = localService.retryRound(round);
            log.info("消费赠券补偿成功，roundLid={}, retryCount={}", round.getLid(), round.getRetryCount());
        } catch (Exception e) {
            log.error("消费赠券补偿失败，roundLid={}", round.getLid(), e);
            // 更新重试次数和下次时间（指数退避）
            roundService.update().lambdaSet()
                .eq(CrmConsumeCouponRound::getRetryCount, round.getRetryCount() + 1)
                .eq(CrmConsumeCouponRound::getNextRetryTime,
                    LocalDateTime.now().plusMinutes((long) Math.pow(2, round.getRetryCount() + 1)))
                .eq(CrmConsumeCouponRound::getErrorMessage, e.getMessage())
                .eq(CrmConsumeCouponRound::getLid, round.getLid()).update();
        }
    }
    log.info("消费赠券补偿任务完成，扫描 {} 条", candidates.size());
}
```

```java
public class CrmConsumeCouponRuleMatcher {
    public CouponMatchResult match(ConsumeCouponGrantRequest request) {
        // 1. 查规则：按 mid + sid + 当前时间过滤
        List<CrmConsumptionCouponRule> rules = ruleService.matchEffective(request.getMid(), request.getSid());
        // 2. 筛选：金额门槛 + 周几限制 + 时段限制 + 会员卡类型/等级
        CrmConsumptionCouponRule matched = rules.stream()
            .filter(r -> matchesAmount(request, r))
            .filter(r -> matchesWeekday(request, r))
            .filter(r -> matchesTimeSlot(request, r))
            .filter(r -> matchesCardType(request, r))
            .findFirst()
            .orElse(null);
        if (matched == null) {
            return CouponMatchResult.noMatch();
        }
        // 3. 计算应发券数量（满额梯度、递增、单次上限）
        CouponPlan plan = calculatePlan(request, matched);
        // 4. 保存规则快照（JSON）— 补偿重试用快照，不重新匹配
        String ruleSnapshot = JSON.toJSONString(new RuleSnapshot(matched));
        return new CouponMatchResult(matched, plan, ruleSnapshot);
    }
}
```

```java
// CrmConsumeCouponRevokeServicePlus.java
public RevokeResult revoke(CrmConsumeBenefitTask task, String strategy) {
    List<CrmConsumeCouponGrantDetail> details = detailService.listByTaskLid(task.getLid());
    List<CrmConsumeCouponGrantDetail> unused = details.stream()
        .filter(d -> "SUCCEEDED".equals(d.getGrantStatus()))
        .filter(d -> !"USED".equals(d.getRevokeStatus()))
        .toList();
    List<CrmConsumeCouponGrantDetail> used = details.stream()
        .filter(d -> "SUCCEEDED".equals(d.getGrantStatus()))
        .filter(d -> "USED".equals(d.getRevokeStatus()))
        .toList();

    // 未使用券：直接回收
    for (CrmConsumeCouponGrantDetail d : unused) {
        couponOpServicePlus.couponOpRevoke(d.getCouponOrderLid(), "CONSUME_REVOKE");
        d.setRevokeStatus("REVOKED");
        detailService.updateById(d);
    }

    // 已使用券：按策略处理
    if (!used.isEmpty()) {
        return handleUsedCoupon(used, strategy, task);
    }
    return RevokeResult.allRevoked(unused.size(), 0);
}

private RevokeResult handleUsedCoupon(List<CrmConsumeCouponGrantDetail> used, String strategy, CrmConsumeBenefitTask task) {
    switch (strategy) {
        case "BLOCK":
            // 阻断：抛业务异常，POS 层拦截
            throw new BizException("消费赠券已被使用，不允许反结账");
        case "EXCEPTION":
            // 异常单：生成 CONSUME_COUPON_REVOKE_EXCEPTION 异常单，继续撤销
            exceptionOrderService.create("CONSUME_COUPON_REVOKE_EXCEPTION", task, used);
            for (CrmConsumeCouponGrantDetail d : used) {
                d.setRevokeStatus("USED_CANNOT_REVOKE");
                detailService.updateById(d);
            }
            return RevokeResult.partialRevoked(unused.size(), used.size());
        case "PARTIAL_COMPENSATE":
            // 补偿：未用回收，已用生成负向权益记录
            // ...实现负向权益逻辑
            return RevokeResult.partialRevoked(unused.size(), used.size());
        default:
            throw new BizException("未知的撤销策略: " + strategy);
    }
}
```

```java
// CrmConsumeCouponLocalService.java
@Service
@Slf4j
public class CrmConsumeCouponLocalService {
    @Transactional(rollbackFor = Exception.class)
    public CrmConsumeCouponSession createSessionIfAbsent(OrderBill order) {
        return sessionService.getOrCreate(order);
    }

    @Transactional(rollbackFor = Exception.class)
    public CrmConsumeCouponRound grantCoupon(CrmConsumeCouponSession session, GrantParams params) {
        // 1. 幂等创建 round
        CrmConsumeCouponRound round = roundService.createRoundIfAbsent(session, params);
        if (round == null) {
            return null; // 已存在，幂等跳过
        }
        // 2. 记录事件
        eventService.logPending(round.getLid(), "GRANT");
        // 3. 发送 ActiveMQ 消息（降级：直调 CRM）
        try {
            activeMQTemplate.convertAndSend("CRM_CONSUME_COUPON_GRANT_QUEUE", toMsg(round));
        } catch (Exception e) {
            log.warn("ActiveMQ 发送失败，降级 Feign", e);
            // 降级：直接调 CRM（走 Feign，同步等待）
            NmsResult<String> result = crmConsumeCouponFeign.consumerGiveCoupon(toDTO(round));
            if (result.isSuccess()) {
                roundService.markGranted(round.getLid(), extractTaskLid(result));
                eventService.logSuccess(round.getLid(), "GRANT", result);
            } else {
                roundService.markFailed(round.getLid(), result.getMsg());
                eventService.logFailed(round.getLid(), "GRANT", result.getMsg());
            }
        }
        return round;
    }

    @Transactional(rollbackFor = Exception.class)
    public void revokeCoupon(CrmConsumeCouponSession session, CrmConsumeCouponRound round, String strategy) {
        // 幂等检查
        if ("REVOKED".equals(round.getStatus()) || "REVOKE_PENDING".equals(round.getStatus())) {
            return; // 已撤销或撤销中，幂等跳过
        }
        // 按策略撤销
        revokeService.revoke(round, strategy);
    }

    // 重试（补偿任务调用）
    public void retryRound(CrmConsumeCouponRound round) {
        if ("GRANTED".equals(round.getStatus())) {
            return; // 已成功，跳过
        }
        // 重新发送消息（幂等键不变）
        activeMQTemplate.convertAndSend("CRM_CONSUME_COUPON_GRANT_QUEUE", toMsg(round));
    }
}
```

## 关键设计决策记录

1. **ActiveMQ 本地缓冲 vs 直接 Feign**：选择本地缓冲（可靠）+ 降级 Feign（容灾），避免同步等待 CRM 超时
2. **补偿任务同时扫 PENDING + FAILED**：PENDING 处理消费者崩溃场景（round 未更新但 CRM 已处理），FAILED 处理 CRM 返回失败场景
3. **幂等键不含 delivery_source**：与统一底座设计文档一致，`originSource = NORMAL` 时不含 delivery_source
4. **规则快照用 JSON 存 rule_snapshot 字段**：补偿重试和反结账都用快照，不用当前规则重算
5. **DLQ 5 次重试后进入死信**：配合告警，人工重投或排查
6. **已使用券默认 BLOCK**：POS 当场反结账优先财务一致性，线上退款默认 EXCEPTION
