## 1. 组件库概述

### 1.1 组件分类

| 分类 | 数量 | 说明 |
|------|------|------|
| 基础表单组件 | 11 | 常用的表单输入组件 |
| 布局容器组件 | 6 | 用于页面布局的容器 |
| 高级输入组件 | 8 | 复杂的输入交互组件 |
| 数据展示组件 | 4 | 用于展示数据的组件 |
| 业务组件 | 3 | 特定业务场景组件 |
| 辅助组件 | 5 | 辅助性UI组件 |
| **总计** | **37** | |

### 1.2 组件通用属性

所有表单组件都支持以下通用属性：

```typescript
interface CommonFieldProps {
  // 基础属性
  id: string;                    // 组件唯一ID
  type: string;                  // 组件类型
  key: string;                   // 字段标识
  label?: string;                // 标签文本
  
  // 显示控制
  display?: {
    visible?: boolean | string;  // 是否显示（支持表达式）
    hidden?: boolean | string;   // 是否隐藏
    disabled?: boolean | string; // 是否禁用
    readonly?: boolean | string; // 是否只读
  };
  
  // 校验规则
  rules?: ValidationRule[];
  
  // 样式
  style?: CSSProperties;
  className?: string;
}
```

## 2. 基础表单组件

### 2.1 Input - 单行输入框

**用途**：文本输入

**属性配置**：

```typescript
interface InputProps extends CommonFieldProps {
  placeholder?: string;          // 占位符
  defaultValue?: string;         // 默认值
  maxLength?: number;            // 最大长度
  minLength?: number;            // 最小长度
  allowClear?: boolean;          // 允许清除
  prefix?: ReactNode;            // 前缀图标
  suffix?: ReactNode;            // 后缀图标
  addonBefore?: string;          // 前置标签
  addonAfter?: string;           // 后置标签
  size?: 'large' | 'default' | 'small';  // 尺寸
}
```

**事件**：
- `onChange(value: string)` - 值变化
- `onBlur()` - 失去焦点
- `onFocus()` - 获得焦点
- `onPressEnter()` - 按下回车

**注册元数据**：

```typescript
export const InputMeta: ComponentMeta = {
  type: 'input',
  name: '单行输入',
  category: ComponentCategory.BASIC,
  icon: <InputOutlined />,
  order: 1,
  defaultSchema: {
    id: '',
    type: 'input',
    key: '',
    label: '输入框',
    props: {
      placeholder: '请输入',
      allowClear: true,
    },
  },
  propsSchema: [
    { key: 'label', label: '标题', type: 'input', category: 'basic', required: true },
    { key: 'key', label: '字段标识', type: 'input', category: 'basic', required: true },
    { key: 'placeholder', label: '占位符', type: 'input', category: 'basic' },
    { key: 'defaultValue', label: '默认值', type: 'input', category: 'basic' },
    { key: 'maxLength', label: '最大长度', type: 'number', category: 'advanced' },
    { key: 'allowClear', label: '允许清除', type: 'switch', category: 'advanced' },
  ],
  events: ['onChange', 'onBlur', 'onFocus', 'onPressEnter'],
  component: InputComponent,
};
```

### 8.2 组件实现模板

```tsx
// Input.tsx
import React from 'react';
import { Input as AntInput } from 'antd';
import { FieldComponentProps } from '@form-designer/shared';

export interface InputProps extends FieldComponentProps {
  placeholder?: string;
  maxLength?: number;
  // ... 其他属性
}

export const Input: React.FC<InputProps> = ({
  value,
  onChange,
  placeholder,
  maxLength,
  disabled,
  readonly,
  ...rest
}) => {
  return (
    <AntInput
      value={value}
      onChange={(e) => onChange?.(e.target.value)}
      placeholder={placeholder}
      maxLength={maxLength}
      disabled={disabled || readonly}
      {...rest}
    />
  );
};
```

### 8.4 测试规范

```tsx
// Input.test.tsx
import { render, fireEvent } from '@testing-library/react';
import { Input } from './Input';

describe('Input Component', () => {
  it('renders correctly', () => {
    const { getByPlaceholderText } = render(
      <Input placeholder="请输入" />
    );
    expect(getByPlaceholderText('请输入')).toBeInTheDocument();
  });

  it('triggers onChange', () => {
    const handleChange = jest.fn();
    const { getByPlaceholderText } = render(
      <Input placeholder="请输入" onChange={handleChange} />
    );
    
    fireEvent.change(getByPlaceholderText('请输入'), {
      target: { value: 'test' },
    });
    
    expect(handleChange).toHaveBeenCalledWith('test');
  });
});
```
