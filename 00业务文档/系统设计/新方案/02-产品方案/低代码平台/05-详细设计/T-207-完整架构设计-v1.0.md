## 一、整体架构

### 1.1 分层架构

```
┌──────────────────────────────────────────────────────┐
│                   Presentation Layer                  │  表现层
│  Designer (设计器) + Renderer (渲染器)                │
└──────────────────────────────────────────────────────┘
                          ↓
┌──────────────────────────────────────────────────────┐
│                      Model Layer                      │  模型层
│  - Form Model (表单模型)                              │
│  - Field Model (字段模型)                             │
│  - Validator Model (校验模型)                         │
│  - Reaction Model (联动模型)                          │
└──────────────────────────────────────────────────────┘
                          ↓
┌──────────────────────────────────────────────────────┐
│                     Metadata Layer                    │  元数据层
│  - Business Object (业务对象)                         │
│  - Form Definition (表单定义)                         │
│  - Component Registry (组件注册)                      │
└──────────────────────────────────────────────────────┘
                          ↓
┌──────────────────────────────────────────────────────┐
│                      Engine Layer                     │  引擎层
│  - Path System (路径系统)                             │
│  - Expression Engine (表达式引擎)                     │
│  - Dependency Tracker (依赖追踪)                      │
│  - Validator Engine (校验引擎)                        │
└──────────────────────────────────────────────────────┘
```

### 1.2 核心概念关系

```
Business Object (BO)                    元数据定义
    ↓
Form Definition                         表单定义（BO的展现形式）
    ↓
Form Model (Runtime)                    运行时表单模型
    ↓
Field Model × N                         字段模型集合
    ↓
Component (View)                        UI组件
```

## 二、元数据驱动架构

### 2.1 业务对象（Business Object）

```typescript
interface BusinessObject {
  boId: string;                    // BO唯一标识
  boName: string;                  // BO名称
  boType: 'entity' | 'view';       // BO类型
  version: string;                 // 版本号
  
  // 实体定义
  entity: {
    tableName: string;             // 数据表名
    primaryKey: string;            // 主键
    displayField: string;          // 显示字段
  };
  
  // 字段定义
  fields: BOField[];
  
  // 关系定义
  relations: BORelation[];
  
  // 业务规则
  rules: BusinessRule[];
  
  // 权限定义
  permissions: PermissionRule[];
}

interface BOField {
  fieldId: string;
  fieldName: string;
  fieldType: 'string' | 'number' | 'date' | 'boolean' | 'reference' | 'calculated';
  dataType: string;                // 数据类型
  length?: number;                 // 长度
  precision?: number;              // 精度
  required: boolean;               // 必填
  unique: boolean;                 // 唯一
  defaultValue?: any;              // 默认值
  
  // 计算字段
  formula?: string;                // 计算公式
  
  // 引用字段
  refBO?: string;                  // 引用的BO
  refField?: string;               // 引用的字段
  
  // 校验规则
  validations?: ValidationRule[];
}

interface BORelation {
  relationId: string;
  relationType: 'one-to-one' | 'one-to-many' | 'many-to-many';
  sourceBO: string;
  targetBO: string;
  sourceField: string;
  targetField: string;
  cascadeDelete?: boolean;
}

interface BusinessRule {
  ruleId: string;
  ruleName: string;
  ruleType: 'before-save' | 'after-save' | 'before-delete' | 'computed';
  condition?: string;              // 触发条件
  action: string;                  // 执行动作
}
```

### 2.2 表单定义（Form Definition）

表单是BO的展现形式之一：

```typescript
interface FormDefinition {
  formId: string;
  formName: string;
  boId: string;                    // 关联的BO
  formType: 'create' | 'edit' | 'view' | 'query';
  version: string;
  
  // 布局定义
  layout: FormLayout;
  
  // 字段绑定
  fields: FormField[];
  
  // 工具栏
  toolbar?: ToolbarConfig;
  
  // 权限控制
  permissions?: FormPermission;
}

interface FormField {
  fieldId: string;
  boField: string;                 // 绑定的BO字段
  componentType: string;           // UI组件类型
  
  // 显示配置
  label?: string;
  placeholder?: string;
  helpText?: string;
  
  // 组件属性
  componentProps?: Record<string, any>;
  
  // 显示控制
  visible?: boolean | string;      // 支持表达式
  disabled?: boolean | string;
  readonly?: boolean | string;
  
  // 校验规则（继承BO + 表单级扩展）
  validations?: ValidationRule[];
  
  // 联动规则
  reactions?: Reaction[];
}
```

**关键设计**：
- ✅ **BO与Form分离**：BO定义业务逻辑，Form定义展现形式
- ✅ **一个BO可以有多个Form**：新建表单、编辑表单、查询表单等
- ✅ **Form绑定BO字段**：`boField`指向BO的字段定义
- ✅ **表单级校验扩展**：可以在Form层增加额外的校验规则

## 三、MVVM架构设计

### 3.1 Model层：Form Model

```typescript
class FormModel {
  private formId: string;
  private boId: string;
  private fields: Map<string, FieldModel> = new Map();
  private dependencyTracker: DependencyTracker;
  private validatorEngine: ValidatorEngine;
  private eventBus: EventBus;
  
  constructor(definition: FormDefinition, bo: BusinessObject) {
    this.formId = definition.formId;
    this.boId = definition.boId;
    
    // 初始化字段模型
    definition.fields.forEach(fieldDef => {
      const boField = bo.fields.find(f => f.fieldId === fieldDef.boField);
      const fieldModel = new FieldModel(fieldDef, boField, this);
      this.fields.set(fieldDef.fieldId, fieldModel);
    });
    
    // 初始化依赖追踪
    this.dependencyTracker = new DependencyTracker();
    this.registerDependencies();
  }
  
  // 获取字段模型（支持路径）
  getField(path: string): FieldModel | undefined {
    return this.fields.get(path);
  }
  
  // 设置字段值
  setFieldValue(path: string, value: any): void {
    const field = this.getField(path);
    if (field) {
      field.setValue(value);
      
      // 触发依赖字段更新
      const affectedFields = this.dependencyTracker.getAffectedFields(path);
      affectedFields.forEach(affectedPath => {
        const affectedField = this.getField(affectedPath);
        affectedField?.recompute();
      });
    }
  }
  
  // 获取表单数据
  getValues(): any {
    const values = {};
    this.fields.forEach((field, path) => {
      setIn(values, path, field.getValue());
    });
    return values;
  }
  
  // 设置表单数据
  setValues(values: any): void {
    this.fields.forEach((field, path) => {
      const value = getIn(values, path);
      if (value !== undefined) {
        field.setValue(value, { silent: true });
      }
    });
  }
  
  // 校验表单
  async validate(): Promise<boolean> {
    const results = await Promise.all(
      Array.from(this.fields.values()).map(field => field.validate())
    );
    return results.every(r => r);
  }
  
  // 重置表单
  reset(): void {
    this.fields.forEach(field => field.reset());
  }
}
```

### 3.2 Model层：Field Model

```typescript
class FieldModel {
  private fieldId: string;
  private boFieldId: string;
  private value: any;
  private initialValue: any;
  private errors: string[] = [];
  
  // 状态
  private visible: boolean = true;
  private disabled: boolean = false;
  private readonly: boolean = false;
  private touched: boolean = false;
  private dirty: boolean = false;
  
  // 配置
  private definition: FormField;
  private boField: BOField;
  private formModel: FormModel;
  
  // 依赖与联动
  private reactions: Reaction[] = [];
  
  constructor(definition: FormField, boField: BOField, formModel: FormModel) {
    this.fieldId = definition.fieldId;
    this.boFieldId = definition.boField;
    this.definition = definition;
    this.boField = boField;
    this.formModel = formModel;
    
    // 初始化默认值
    this.initialValue = boField.defaultValue;
    this.value = this.initialValue;
    
    // 初始化联动规则
    this.reactions = definition.reactions || [];
  }
  
  // 设置值
  setValue(value: any, options?: { silent?: boolean }): void {
    const oldValue = this.value;
    this.value = value;
    this.dirty = value !== this.initialValue;
    
    if (!options?.silent) {
      this.formModel.eventBus.emit('field:change', {
        fieldId: this.fieldId,
        oldValue,
        newValue: value,
      });
    }
    
    // 触发联动
    this.executeReactions();
  }
  
  // 获取值
  getValue(): any {
    return this.value;
  }
  
  // 校验
  async validate(): Promise<boolean> {
    this.errors = [];
    
    // BO级校验
    if (this.boField.validations) {
      for (const rule of this.boField.validations) {
        const result = await this.formModel.validatorEngine.executeRule(
          rule,
          this.value,
          this.formModel.getValues()
        );
        if (result !== true) {
          this.errors.push(result as string);
        }
      }
    }
    
    // Form级校验
    if (this.definition.validations) {
      for (const rule of this.definition.validations) {
        const result = await this.formModel.validatorEngine.executeRule(
          rule,
          this.value,
          this.formModel.getValues()
        );
        if (result !== true) {
          this.errors.push(result as string);
        }
      }
    }
    
    return this.errors.length === 0;
  }
  
  // 重新计算（用于计算字段）
  recompute(): void {
    if (this.boField.fieldType === 'calculated' && this.boField.formula) {
      const computed = this.formModel.expressionEngine.execute(
        this.boField.formula,
        this.formModel.getValues()
      );
      this.setValue(computed, { silent: true });
    }
  }
  
  // 执行联动规则
  private executeReactions(): void {
    this.reactions.forEach(reaction => {
      if (reaction.condition) {
        const conditionMet = this.formModel.expressionEngine.execute(
          reaction.condition,
          this.formModel.getValues()
        );
        if (!conditionMet) return;
      }
      
      // 执行动作
      if (reaction.target && reaction.action) {
        const targetField = this.formModel.getField(reaction.target);
        if (targetField) {
          switch (reaction.action) {
            case 'set-value':
              targetField.setValue(reaction.value);
              break;
            case 'set-visible':
              targetField.setVisible(reaction.value);
              break;
            case 'set-disabled':
              targetField.setDisabled(reaction.value);
              break;
          }
        }
      }
    });
  }
  
  // 重置
  reset(): void {
    this.value = this.initialValue;
    this.errors = [];
    this.touched = false;
    this.dirty = false;
  }
}
```
