### 1. 架构创新

✅ **MVVM架构**
- Field Model：字段级MVVM
- Form Model：表单级MVVM
- SubTable Model：子表级MVVM
- MobX响应式绑定

✅ **元数据驱动**
- BO层：业务逻辑定义
- Form层：UI展现定义
- 双层分离，职责清晰

✅ **路径系统**
- 支持嵌套对象访问
- 支持数组索引访问
- 通配符路径（items.*.amount）

✅ **表达式引擎**
- 字段引用
- 内置函数（SUM/AVG/MAX/MIN等）

- 依赖分析

- 反向索引（快速查找受影响字段）
- 拓扑排序（计算顺序）
- 循环依赖检测

import { FormModel, BusinessObject, FormDefinition } from '@lowcode/form-core';
import { FormRenderer } from '@lowcode/form-renderer';
import { FormDesigner } from '@lowcode/form-designer';

// 1. 定义业务对象
const OrderBO: BusinessObject = { ... };

// 2. 定义表单
const OrderForm: FormDefinition = { ... };

// 3. 使用渲染器
const formModel = new FormModel(OrderForm, OrderBO);
<FormRenderer formModel={formModel} onSubmit={handleSubmit} />

// 4. 使用设计器
<FormDesigner bo={OrderBO} onSave={handleSave} />
